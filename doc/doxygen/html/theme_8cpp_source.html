51  // If we cannot open the theme, we load the default valued

54  useFallbackColors();

58  // These strings will be used to deternmine

60  QStringList _types;

62 
63  // Add all types and colors in their respective strings
64  QXmlStreamReader *_xml_reader = new QXmlStreamReader(&_file);
65  while (!_xml_reader->atEnd() && !_xml_reader->hasError()) {
66  QXmlStreamReader::TokenType token = _xml_reader->readNext();
67  if (token == QXmlStreamReader::StartElement) {
68  if (_xml_reader->name() == "type")
69  _types.append(_xml_reader->readElementText());
70  if (_xml_reader->name() == "color")

72  }

78 
79  // Only continue if the XML file is properly written

85  // Set the colors of the application
