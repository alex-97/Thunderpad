159 void ToolBar::updateSettings() {

162 
163  // Hide/show the toolbar
164  setVisible(m_settings->value("toolbar-enabled", true).toBool());

166 
167  // Update the toolbar style, we use the "if" condition to avoid
168  // resizing and redrawing the toolbar each time that the user changes
169  bool _new_value = m_settings->value("toolbar-text", MAC_OS_X).toBool();
170  if (_new_value != m_old_value) {
171  setToolbarText(_new_value);
172  m_old_value = _new_value;
173  }
