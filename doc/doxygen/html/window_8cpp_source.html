382 void Window::updateTitle() {
383  QString _title;
384 
385  // Use "untitled" when we have no saved document
386  if (m_editor->documentTitle().isEmpty()) _title = tr("Untitled");
387 
388  // Use the file name in the title
389  else
390  _title = shortFileName(m_editor->documentTitle());
391 
392  // Add a "*" if the document was modifed
393  if (m_editor->document()->isModified())
394  setWindowTitle(_title + "* - Thunderpad");
395  else
396  setWindowTitle(_title + " - Thunderpad");
397 
398  // Configure the behavior of the 'smart' save actions
399  if (!m_editor->documentTitle().isEmpty() &&
400  !m_editor->document()->isModified()) {
401  m_menu->setSaveEnabled(false);
402  m_toolbar->setSaveEnabled(false);
403  } else {
404  m_menu->setSaveEnabled(true);
405  m_toolbar->setSaveEnabled(true);
406  }
407 }

493 void Window::configureWindow(Window *window) {
494  // Save the state of the window
495  window->saveWindowState();
496 
497  // Ensure that all windows are connected together and synced together
498  foreach(QWidget * _widget, QApplication::topLevelWidgets()) {
499  if (_widget->objectName() == "window") {
500  connect(_widget, SIGNAL(settingsChanged()), window,
501  SIGNAL(updateSettings()));
502  connect(window, SIGNAL(settingsChanged()), _widget,
503  SIGNAL(updateSettings()));
504  }
505  }
506 
507  // Reposition the window
508  window->resize(size());
509  window->move(window->x() + 45, window->y() + 45);
510  m_settings->setValue("position", QPoint(window->x(), window->y()));
511 }
