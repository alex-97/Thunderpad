156 void Editor::updateSettings() {

160  // operating system
161  int _default_size = 11;
162  QString _default_font = "Courier";
163 
164  // Use Menlo (as in Xcode) in Mac OS X

180  // does not support saving fonts directly

184  _settings_font.setPointSize(

186  _settings_font.setBold(_settings.value("font-bold", false).toBool());
187  _settings_font.setItalic(_settings.value("font-italic", false).toBool());

194

197  QPalette _palette = palette();

199  _palette.setColor(QPalette::Text, m_theme->foreground());

206  highlightCurrentLine();

249 bool Editor::maybeSave() {
250  // We don't need to save the document if it isn't modified
251  if (!document()->isModified()) return true;
252 
253  // The document was already saved in the hard disk, however,
254  // it has unsaved changes.
255  if (!documentTitle().isEmpty() && document()->isModified()) return save();
256 
257  // The document was never saved as a file, so we need to prompt
258  // the user where to save our document.
259  else if (document()->isModified()) {
260  QMessageBox _message;
261  _message.setParent(this);
262  _message.setWindowModality(Qt::WindowModal);
263  _message.setWindowIcon(QIcon(":/icons/dummy.png"));
264  _message.setIconPixmap(QPixmap(":/icons/logo.png"));
265  _message.setStandardButtons(QMessageBox::Save | QMessageBox::Cancel |
266  QMessageBox::Discard);
267  _message.setText(
268  "<b>" + tr("This document has changes, do you want to save them?") +
269  "</b>");
270 
271  _message.setInformativeText(
272  tr("Your changes will be lost if you close this item without saving."));
273 
274  int _return_ = _message.exec();
275  if (_return_ == QMessageBox::Save)
276  return save();
277  else if (_return_ == QMessageBox::Discard)
278  return true;
279  else
280  return false;
281  }
282 
283  return false;
284 }
